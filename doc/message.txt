Для подключения ресивера всего несколько строк нужно
import "gopkg.in/webnice/log.v2"
import r "gopkg.in/webnice/log.v2/receiver"
import s "gopkg.in/webnice/log.v2/sender"

var receiver s.Receiver
receiver = r.SyslogReceiver.Receiver
s.Gist().AddSender(receiver)
